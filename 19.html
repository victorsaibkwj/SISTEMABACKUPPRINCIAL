Sobre o Projeto

O Robô Sushuman é um braço robótico automatizado para preparar alimentos, como sushi, controlado por Arduino e servomotores. Ideal para projetos de automação caseira e fins educacionais.

Hardware Necessário

  • Arduino Mega 2560
  • 16x Servomotores MG996R
  • PCA9685 - Controlador PWM
  • Fonte 5V 5A
  • Estrutura impressa em 3D
  • Sensor ultrassônico (opcional)

using System;
using System.Threading;

class RoboSushuman
{
    static void Main()
    {
        for (int i = 0; i < 2; i++)
        {
            MoveServo(0, 90);
            Thread.Sleep(1000);
            MoveServo(0, 0);
            Thread.Sleep(1000);
        }
    }

    static void MoveServo(int canal, int grau)
    {
        int pulso = Map(grau, 0, 180, 150, 600);
        Console.WriteLine($"Movendo servo {canal} para grau {grau} (pulso: {pulso})");
    }

    static int Map(int valor, int inMin, int inMax, int outMin, int outMax)
    {
        return (valor - inMin) * (outMax - outMin) / (inMax - inMin) + outMin;
    }
}

Trecho de Código Arduino


#include <Wire.h>
#include <Adafruit_PWMServoDriver.h>

Adafruit_PWMServoDriver pwm = Adafruit_PWMServoDriver();

void setup() {
  pwm.begin();
  pwm.setPWMFreq(60);
}

void moveServo(uint8_t canal, uint16_t grau) {
  uint16_t pulso = map(grau, 0, 180, 150, 600);
  pwm.setPWM(canal, 0, pulso);
}

void loop() {
  moveServo(0, 90);
  delay(1000);
  moveServo(0, 0);
  delay(1000);
}
